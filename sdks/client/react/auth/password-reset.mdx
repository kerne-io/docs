<Note>
The React SDK does not currently expose password reset methods directly. These should be handled through API routes on your server using the Server SDK.
</Note>

## Server-side handling

Use the [Server SDK auth methods](/sdks/server/auth/password-reset) in your API routes:

```typescript
// /api/auth/request-reset
import { kerne } from '@/lib/kerne';

export async function POST(req: Request) {
  const { email } = await req.json();
  await kerne.auth.requestPasswordReset(email);
  return Response.json({ success: true });
}
```

```typescript
// /api/auth/reset-password
import { kerne } from '@/lib/kerne';

export async function POST(req: Request) {
  const { token, password } = await req.json();
  await kerne.auth.confirmPasswordReset(token, password);
  return Response.json({ success: true });
}
```

## Client-side flow

### Step 1: Request reset

```tsx
import { useState } from 'react';

function ForgotPasswordForm() {
  const [email, setEmail] = useState('');
  const [sent, setSent] = useState(false);

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();
    // ...
    await fetch('/api/auth/request-reset', {
      method: 'POST',
      body: JSON.stringify({ email })
    });
    setSent(true);
    // ...
  };

  // ...
}
```

### Step 2: Reset with token

```tsx
import { useState } from 'react';
import { useSearchParams } from 'next/navigation';

function ResetPasswordForm() {
  const searchParams = useSearchParams();
  const token = searchParams.get('token');
  const [password, setPassword] = useState('');

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();
    // ...
    await fetch('/api/auth/reset-password', {
      method: 'POST',
      body: JSON.stringify({ token, password })
    });
    // ...
  };

  // ...
}
```
