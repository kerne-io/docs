Create a new project through the SDK client available via the `useClient` hook.

## Using the hook

```tsx
import { useClient } from '@kerne/react';

function CreateProject() {
  const kerne = useClient();
  
  const project = await kerne.projects.create({ name: 'My Project' });
  // ...
}
```

## Parameters

| Name | Type | Required | Description |
|------|------|----------|-------------|
| `name` | `string` | Yes | Project name |
| `slug` | `string` | No | URL-friendly identifier (auto-generated if not provided) |
| `description` | `string` | No | Project description |

## Response

Returns the created `Project` object.

```typescript
interface Project {
  id: string;
  name: string;
  slug: string;
  description?: string;
  created_at: string;
}
```

## Full example

```tsx
import { useState } from 'react';
import { useClient } from '@kerne/react';

function CreateProjectForm() {
  const kerne = useClient();
  const [name, setName] = useState('');
  const [loading, setLoading] = useState(false);

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();
    setLoading(true);
    
    try {
      const response = await kerne.projects.create({ name });
      console.log('Project created:', response.data);
      window.location.href = `/projects/${response.data.id}`;
    } catch (error) {
      console.error('Failed to create project:', error);
    } finally {
      setLoading(false);
    }
  };

  return (
    <form onSubmit={handleSubmit}>
      <input
        type="text"
        value={name}
        onChange={(e) => setName(e.target.value)}
        placeholder="Project name"
        required
      />
      <button type="submit" disabled={loading}>
        {loading ? 'Creating...' : 'Create project'}
      </button>
    </form>
  );
}
```
