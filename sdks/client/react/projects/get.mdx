---
title: "Get project"
description: "Retrieve a single project"
---

Retrieve a project by ID through the SDK client available via the `useClient` hook.

## Using the hook

```tsx
import { useClient } from '@kerne/react';

function ProjectDetail({ id }) {
  const kerne = useClient();
  
  const project = await kerne.projects.get(id);
  // ...
}
```

## Parameters

| Name | Type | Description |
|------|------|-------------|
| `id` | `string` | Project ID |

## Response

Returns a `Project` object.

## Full example

```tsx
import { useState, useEffect } from 'react';
import { useClient } from '@kerne/react';
import { useParams } from 'react-router-dom';

function ProjectPage() {
  const { id } = useParams();
  const kerne = useClient();
  const [project, setProject] = useState(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    async function loadProject() {
      try {
        const response = await kerne.projects.get(id);
        setProject(response.data);
      } catch (error) {
        console.error('Failed to load project:', error);
      } finally {
        setLoading(false);
      }
    }
    
    loadProject();
  }, [id, kerne]);

  if (loading) return <p>Loading...</p>;
  if (!project) return <p>Project not found</p>;

  return (
    <div>
      <h1>{project.name}</h1>
      <p>{project.description}</p>
    </div>
  );
}
```
